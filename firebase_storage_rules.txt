// Firebase Storage Rules untuk QuizMate App
// Paste rules ini ke Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Rules for quiz materials
    match /quiz_materials/{allPaths=**} {
      // Allow read for authenticated users
      allow read: if request.auth != null;
      // Allow write for authenticated users
      allow write: if request.auth != null 
                   && request.auth.uid != null
                   && resource == null; // Only allow creating new files
    }
    
    // Rules for profile pictures
    match /profile_pictures/{userId}/{allPaths=**} {
      // Allow read for authenticated users
      allow read: if request.auth != null;
      // Allow write only for the user's own profile
      allow write: if request.auth != null 
                   && request.auth.uid == userId;
    }
    
    // Default rule - deny everything else
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
